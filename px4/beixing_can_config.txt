unsigned long CrcGen_STM32(unsigned long data[], unsigned long size)
{
	unsigned long i,j,temp,crc = 0xFFFFFFFF;
	for(i=0; i<size; i++)
	{
	temp = data[i];
	for(j=0; j<32; j++)
	{
	if( (crc ^ temp) & 0x80000000 )
	{
	crc = 0x04C11DB7 ^ (crc<<1);
	}
	else
	{
	crc <<=1;
	}
	temp<<=1;
	}
	}
	return crc;
}
